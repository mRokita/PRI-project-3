123 int
124 cmp(Animal* a, Animal* b){
125  int res = 0;
126  if(!sort_asc) {
127  Animal* t = a;
128  a = b;
129  b = t;
130  }
131  if(sort_by == ID) res = a->id > b->id;
132  if(sort_by == AGE) res = a->age > b->age;
133  if(sort_by == NAME) res = strcmp(a->name, b->name) > 0;
134  if(sort_by == SPECIES) res = strcmp(a->species, b->species) > 0;
135  if(sort_by == COMMENT) res = strcmp(a->comment, b->comment) > 0;
136  return res;
137 }
